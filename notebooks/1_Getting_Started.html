Getting Started¶

[1]:

%load_ext autoreload
%autoreload 2

[2]:

from metricx import Metric, Task

[3]:

task1 = Task(
    name="task-1",
    metrics=[
        Metric(name="score", is_higher_better=True),
        Metric(name="fit-time", is_higher_better=False),
    ],
)
task1.report("model-A", {"score": 1.0, "fit-time": 0.1})
task1.report("model-A", {"score": 1.1, "fit-time": 0.2})
task1.report("model-B", {"score": 0.0, "fit-time": 0.4})
task1.report("model-C", {"score": 0.1, "fit-time": 0.5})
task1.to_df()
